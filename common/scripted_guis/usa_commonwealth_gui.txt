scripted_gui = {
    USA_commonwealth_button_gui = {
        window_name = "USA_commonwealth_button_gui_window"
        context_type = player_context
        parent_window_token = politics_tab
         
        visible = {
            tag = USA
        }
        
        effects = {
			Button_1_click = {
				if = {
					limit = {
						has_country_flag = USA_commonwealth_gui_open
					}
					clr_country_flag = USA_commonwealth_gui_open

					set_variable = { USA_commonwealth_gui_tab = 0 }
				}
				else = {
					set_country_flag = USA_commonwealth_gui_open
				}
			}
        }
        
        triggers = {
        }
        
        properties = {
        }
        
    }
}
