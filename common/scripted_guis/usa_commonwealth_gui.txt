scripted_gui = {
    USA_commonwealth_button_gui = {
        window_name = "USA_commonwealth_button_gui_window"
        context_type = player_context
        parent_window_token = politics_tab
         
        visible = {
            tag = USA
        }
        
        effects = {
			Button_1_click = {
				if = {
					limit = {
						has_country_flag = USA_commonwealth_gui_open
					}
					clr_country_flag = USA_commonwealth_gui_open

					set_variable = { tab_var = 0 }
				}
				else = {
					set_country_flag = USA_commonwealth_gui_open
				}
			}
        }
        
        triggers = {
        }
        
        properties = {
        }   
    }
}

scripted_gui = {
    USA_commonwealth_gui = {
        window_name = "USA_commonwealth_gui_window"
        context_type = player_context
         
        visible = {
            has_country_flag = USA_commonwealth_gui_open
        }
        
        effects = { 
			Button_1_click = {
				clr_country_flag = USA_commonwealth_gui_open
			}
			Button_2_click = { ## Tab 1
				set_variable = { 
					var = tab_var
					value = 1
				}
			}
			Button_3_click = { ## Tab 2
				set_variable = { 
					var = tab_var
					value = 2
				}
			}
			Button_4_click = { ## Tab 3
				set_variable = { 
					var = tab_var
					value = 3 
				}
			}
			Button_5_click = { ## Close tabs
				set_variable = { 
					var = tab_var
					value = 0
				}
			}
        }
        
        triggers = {
        }
        
        properties = {
        }

		dynamic_lists = {
			tab_grid_box = {
					array = tab_array

					value = tab_v

					index = tab_i
					
					change_scope = no

					entry_container = "Commonwealth_tab_[?tab_var]"
			}
		}
    }
}
